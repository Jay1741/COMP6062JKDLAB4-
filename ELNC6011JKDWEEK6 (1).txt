/* Preprocessor ===============================================================
   Hardware Configuration Bits ==============================================*/
#pragma config FOSC		= INTIO67
#pragma config PLLCFG	= OFF
#pragma config PRICLKEN = ON
#pragma config FCMEN	= OFF
#pragma config IESO		= OFF
#pragma config PWRTEN	= OFF 
#pragma config BOREN	= ON
#pragma config BORV		= 285 
#pragma config WDTEN	= OFF
#pragma config PBADEN	= OFF
#pragma config LVP		= ON
#pragma config MCLRE	= EXTMCLR

// Libraries ==================================================================
#include <xc.h>
#include <stdio.h>
#include <stdlib.h>

// Constants  =================================================================
#define TRUE        1	
#define FALSE       0	
#define SAMPLES     10
#define TEMP        0
#define HUMID       1
#define PRES        2
#define SENSORS     3
#define ADCCHAN     ADCON0bits.CHS 
#define ADCGO       ADCON0bits.GO
#define TX1BUSY     PIR1bits.TX1IF
#define PBPORT      PORTA
#define PBMASK      0xF0
#define READPBS     (PBPORT & PBMASK)
#define TMR0IF      INTCONbits.TMR0IF    
#define ONESECCONT  4
#define PSC         0x0BDC
#define TEMPM       0.0977
#define TEMPB       0
#define HUMIDM      0.0977
#define HUMIDB      0
#define PRESM       1.465
#define PRESB       0
#define ADCRESV     (5/1024)
#define NOPRESS     0xF0
#define CHANSEL     0xE0
#define MODE        0xD0
#define INC         0xB0
#define DEC         0x70
// Type Defs ==========================================================
typedef int sensors_t;
typedef char flag_t;
// Global Variables  ==========================================================
typedef struct 
{
	sensors_t current;
	int highLimit;
	int lowLimit;
	sensors_t samples[SAMPLES];
	int insert;
	flag_t avgRdy;
	
}sensorChannel_t;

sensorChannel_t sensors [ SENSORS ];

typedef struct
{
    char channelSel;   // 0,1,2
    char mode;            // 0 = Low, 1 = High
    char pbState;
    char lastState;
} pbs_t;

pbs_t pbs;

// Functions  =================================================================

/*>>> setOsc4mhz ===========================================================
Author:		Jay Darji
Date:		15/01/2001
Modified:	None
Desc:		This function sets the frequency at 4Mhz for internal Oscillator of Microcontroller
			then waits it still stabalize.
Input: 		None 
Returns:	None 
 ============================================================================*/
void setOsc4mhz(void)
{
	OSCCON = 0x53;
	while(!OSCCONbits.HFIOFS);
	
} // eo setOsc4mhz::

/*>>> setIOport: ===========================================================
Author:		Jay Darji
Date:		15/01/2026
Modified:	None
Desc:		Configures PORTA for analog inputs and PORTC for USART.
Input: 		None 
Returns:	None 
 ============================================================================*/
void setIOport(void)
{
	ANSELA = 0x07; // RA0,RA1 AND RA2 as analog input.
	LATA   = 0x00;
	TRISA  = 0xFF;
	
	ANSELB = 0x00;
	LATB   = 0x00;
	TRISB  = 0xFF;
	
	ANSELC = 0x00;
	LATC   = 0x00;
	TRISC  = 0xBF; // Set the RC6 pin as output for USART 1
	  
	ANSELD = 0x00;
	LATD   = 0x00;
	TRISD  = 0xFF;
	 
	ANSELE = 0x00;
	LATE   = 0x00;
	TRISE  = 0xFF;
} // setIOport::

/*>>> configAdc: ===========================================================
Author:		Jay Darji
Date:		29/01/2026
Modified:	None
Desc:		Configures ADC for 12 Tad, Fosc/8, and module ON.   
Input: 		None 
Returns:	None 
 ============================================================================*/
void configAdc(void)
{
	ADCON0 = 0x01;
    ADCON1 = 0x00;
    ADCON2 = 0xA9; // right justified,12 tad, fosc/8 
 
} // eo configAdc::

/*>>> getAdcSample: ===========================================================
Author:		Jay Darji
Date:		29/01/2026
Modified:	None
Desc:		This function Selects a channel, starts conversion, and returns the result. 
Input: 		char chan - input for channel select
Returns:	ADRES - the sample value from register.
 ============================================================================*/
int getAdcSample(char chan)
{
	ADCCHAN = chan;
    ADCGO = TRUE;
    while(ADCGO);
    return ADRES;
    
} // eo getAdcSample::

/*>>>  configSp1: ===========================================================
Author:		Jay Darji
Date:		29/01/2026
Modified:	None
Desc:		This function configures the serial connection on port 1 for establishing coomunication 
            with MAX 232.It sets baud rate at  9600, SP1 on, TX & RX enabled, 8 bit, 1 stop bit, non-inverted.
Input: 		None 
Returns:	None 
 ============================================================================*/
void configSp1(void)
{
	TXSTA1 = 0x26;
    RCSTA1 = 0x80;
    BAUDCON1 = 0X40;
    SPBRGH1 = 0X00;
    SPBRG1 = 0X19;
       
} // eo configSp1::

/*>>>  resetTMR0: ===========================================================
Author:		Jay Darji
Date:		12/02/2026
Modified:	None
Desc:		This function reset the timer by putting timer flag to zero State and set the value 
            high resistor and low resistor in TIMER 0
Input: 		int psc,  2 byte value to set the timer on desired rollover  
Returns:	None 
 ============================================================================*/
void resetTMR0(int psc)
{
	TMR0IF = FALSE;
    TMR0H = psc >> 8;
    TMR0L = psc;
       
} // eo resetTMR0::

/*>>>  configTMR0: ===========================================================
Author:		Jay Darji
Date:		12/02/2026
Modified:	None
Desc:		This function configures the timer resistor for Specific rollover and then reset the timer.
Input: 		int psc,  2 byte value to set the timer on desired rollover  
Returns:	None 
 ============================================================================*/
void configTMR0(int psc)
{
	T0CON = 0x91;
    resetTMR0(psc);
    
} // eo configTMR0::

/*>>> systemInitialization: ===========================================================
Author:		Jay Darji
Date:		29/01/2026
Modified:	None
Desc:		This function initialize all the functions for main.
Input: 		None 
Returns:	None 
 ============================================================================*/
void systemInitialization(void)
{
    setOsc4mhz();
    setIOport();
    configAdc();
    configSp1(); 
    configTMR0(PSC);
} // eo systemInitialization::

/*>>>  putch: ===========================================================
Author:		Jay Darji
Date:		05/02/2026
Modified:	None
Desc:		This function Waits for TX1 Flag to be in TRUE state then redirect output to USART1.
Input: 		char data, sets the value of TXREG1 to variable. 
Returns:	None 
 ============================================================================*/
void putch(char data)
{
    while(!TX1BUSY);
    TXREG1= data;
}// eo putch ::

/*>>> printValue: ========================================================
Author:		Jay Darji
Date:		05/01/2026
Modified:	None
Desc:		This function transmits the formatted sensor data table via 
			USART1, including the student ID, current averages, and limits.
Input: 		None 
Returns:	None 
============================================================================*/
void printValue(void)
{
	printf("\033[2J\033[H"); // Clear Screen
	printf("System(933)\n\r");
    printf("Channel: %i ",pbs.channelSel);
    if(pbs.mode)
    {
        printf("Mode: High \n\r");
    }
    else
    {
        printf("Mode: Low \n\r");
    }
    
	printf("Sen0:\t%i  Sen1:\t%i  Sen2:\t%i \n\r", sensors[TEMP].current, sensors[HUMID].current, sensors[PRES].current); // Display Averages	
	printf("HL:  \t%i  HL:\t  %i  HL:\t  %i \n\r", sensors[TEMP].highLimit, sensors[HUMID].highLimit, sensors[PRES].highLimit); // Display High Limits	
	printf("LL:  \t%i  LL:\t  %i  LL:\t  %i \n\r", sensors[TEMP].lowLimit, sensors[HUMID].lowLimit, sensors[PRES].lowLimit); // Display Low Limits
    
} // eo printValue::

/*>>> changeChan: ========================================================
Author:		Jay Darji
Date:		05/01/2026
Modified:	None
Desc:		This function transmits the formatted sensor data table via 
			USART1, including the student ID, current averages, and limits.
Input: 		None 
Returns:	None 
============================================================================*/
void changeChan(void)
{
    pbs.channelSel++;
    if (pbs.channelSel >= SENSORS)
    {
        pbs.channelSel = TEMP;
    }
}// eo changeChan::
/*>>> changeMode ========================================================
Author:		Jay Darji
Date:		05/01/2026
Modified:	None
Desc:		This function transmits the formatted sensor data table via 
			USART1, including the student ID, current averages, and limits.
Input: 		None 
Returns:	None 
============================================================================*/
void changeMode(void)
{
    pbs.mode =! pbs.mode;
}// eo changeMode::

/*>>> incLimit: ========================================================
Author:		Jay Darji
Date:		05/01/2026
Modified:	None
Desc:		This function transmits the formatted sensor data table via 
			USART1, including the student ID, current averages, and limits.
Input: 		None 
Returns:	None 
============================================================================*/
void incLimit(void)
{
    if (pbs.mode)
        sensors[pbs.channelSel].highLimit++;
    else
        sensors[pbs.channelSel].lowLimit++;
}// eo incLimit::
/*>>> decLimit: ========================================================
Author:		Jay Darji
Date:		05/01/2026
Modified:	None
Desc:		This function transmits the formatted sensor data table via 
			USART1, including the student ID, current averages, and limits.
Input: 		None 
Returns:	None 
============================================================================*/
void decLimit(void)
{
    if (pbs.mode)
        sensors[pbs.channelSel].highLimit--;
    else
        sensors[pbs.channelSel].lowLimit--;
}// eo decLimit::

// Sensor Functions  =================================================================

/*>>> initSensorData: ===========================================================
Author:		Jay Darji
Date:		08/01/2026
Modified:	None
Desc:		Function is intialize a data structure sensoreChennel_t by putting it to zero State.
Input: 		sensorChannel_t *ptr, A Pointer to a sensorChannel_t object.
Returns:	None
 ============================================================================*/
void initSensorData(sensorChannel_t *ptr,int chID)
{
	int index = 0;
	ptr -> current = FALSE;
	ptr -> insert = FALSE;
	ptr -> avgRdy = FALSE;
	for (index = 0; index < SAMPLES;index++)
	{
		ptr -> samples[index] = FALSE;
	}
    switch(chID)
    {
        case TEMP:
            ptr -> highLimit = 150;
            ptr -> lowLimit = 50;
            break;
            
        case HUMID:
            ptr -> highLimit = 100;
            ptr -> lowLimit = 0;
            break;    
            
        case PRES:
            ptr -> highLimit = 1500;
            ptr -> lowLimit = 200;
            break;    
            
    }
	
} // eo initSensorData::



/*=== MAIN: FUNCTION ==========================================================
 ============================================================================*/
void main(void)
{
	char chID = 0;
	long sum = 0;
	int index = 0;
	int secCount = 0;
    float calc = 0;
	systemInitialization();
	
	for (chID = 0; chID < SENSORS; chID++)
	{
		initSensorData(&sensors[chID],chID);
	} // eo for loop
    
	while(TRUE)
	{  
        if(TMR0IF)
        {
            resetTMR0(PSC);
            secCount ++;
            if (secCount >= ONESECCONT)
            {
                secCount = 0;
            }// eo if
        }// eo if
		for (chID = 0; chID < SENSORS; chID++)
		{
			sum = 0;
			 
			sensors[chID].samples[sensors[chID].insert] = getAdcSample(chID);
			sensors[chID].insert++;
			
			if(sensors[chID].insert >= SAMPLES)
			{
				sensors[chID].insert = FALSE;
				sensors[chID].avgRdy = TRUE;
			} // eo if
			
			if (sensors[chID].avgRdy == TRUE)
			{
				for(index = 0; index < SAMPLES; index++) 
				{
					sum += sensors[chID].samples[index];
				} // eo for loop
				
				sensors[chID].current = (sum / SAMPLES);
                
                calc = ADCRESV * sensors[chID].current ;
                
                switch(chID)
                {
                    case TEMP:
                        calc = calc - TEMPB;
                        calc = calc/TEMPB;
                        break;
                    
                    case HUMID:
                        calc = calc - HUMIDB;
                        calc = calc/HUMIDB;
                        break;
                        
                    case PRES:
                        calc = calc - PRESB;
                        calc = calc/PRESB;
                        break;                
                } // eo swich        
			} // eo if
		} // eo for loop
		pbs.pbState = READPBS;
        if(pbs.pbState != NOPRESS && pbs.pbState != pbs.lastState)
        {
            pbs.lastState = pbs.pbState;
            switch(pbs.pbState)
            {
                case CHANSEL:
                    changeChan();
                    break;
                
                case MODE:
                    changeMode();
                    break;
                    
                case INC:
                    incLimit();
                    break;
                    
                case DEC:
                    decLimit();
                    break;           
            }//eo switch 
        }// eo if 
        else 
        {
            if (pbs.pbState == NOPRESS)
            {
                pbs.lastState = FALSE;               
            } // eo if 
        }// eo else 
        
		printValue(); 
		
	} // eo while loop
   
} // eo main::
